```c++
#include <iostream>
#include <math.h>
using namespace std;

int main(){
	float s[4]= {1.3, 8.7, 46.5, 11.1};

	for(int i = 0; i<=3; i++){
		cout << roundf(s[i]) <<'\n'; 
	}
	
	return 0;
}
```
